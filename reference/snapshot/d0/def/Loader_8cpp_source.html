145  void ImGuiLoader::newFrame()
146  {
147  _internalNewFrame();
148 
149  ImGuiIO& io = ImGui::GetIO();
150 
151  // Setup display size (every frame to accommodate for window resizing)
152  int w, h;
153  int display_w, display_h;
154  glfwGetWindowSize(_window, &w, &h);
155  glfwGetFramebufferSize(_window, &display_w, &display_h);
156  io.DisplaySize = ImVec2((float)w, (float)h);
157  io.DisplayFramebufferScale = ImVec2(w > 0 ? ((float)display_w / w) : 0, h > 0 ? ((float)display_h / h) : 0);
158 
159  // Setup time step
160  double current_time = glfwGetTime();
161  io.DeltaTime = _time > 0.0 ? (float)(current_time - _time) : (float)(1.0f/60.0f);
162  _time = current_time;
163 
164  // Setup inputs
165  // (we already got mouse wheel, keyboard keys & characters from glfw callbacks polled in glfwPollEvents())
166  if (glfwGetWindowAttrib(_window, GLFW_FOCUSED))
167  if (io.WantMoveMouse)
168  {
169  glfwSetCursorPos(_window, (double)io.MousePos.x, (double)io.MousePos.y); // Set mouse position if requested by io.WantMoveMouse flag (used when io.NavMovesTrue is enabled by user and using directional navigation)
170  }
171  else
172  {
173  double mouse_x, mouse_y;
174  glfwGetCursorPos(_window, &mouse_x, &mouse_y);
175  io.MousePos = ImVec2((float)mouse_x, (float)mouse_y);
176  }
177  else
178  io.MousePos = ImVec2(-FLT_MAX,-FLT_MAX);
179 
180  for (int i = 0; i < 3; i++)
181  {
182  // If a mouse press event came, always pass it as "mouse held this frame", so we don't miss click-release events that are shorter than 1 frame.
183  io.MouseDown[i] = _mousePressed[i] || glfwGetMouseButton(_window, i) != 0;
184  _mousePressed[i] = false;
185  }
186 
187  // Update OS/hardware mouse cursor if imgui isn't drawing a software cursor
188  ImGuiMouseCursor cursor = ImGui::GetMouseCursor();
189  if (io.MouseDrawCursor || cursor == ImGuiMouseCursor_None)
190  {
191  glfwSetInputMode(_window, GLFW_CURSOR, GLFW_CURSOR_HIDDEN);
192  }
193  else
194  {
195  glfwSetCursor(_window, _mouseCursors[cursor] ? _mouseCursors[cursor] : _mouseCursors[ImGuiMouseCursor_Arrow]);
196  glfwSetInputMode(_window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
197  }
198 
199  // Gamepad navigation mapping [BETA]
200  memset(io.NavInputs, 0, sizeof(io.NavInputs));
201  if (io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad)
202  {
203  // Update gamepad inputs
204  #define MAP_BUTTON(NAV_NO, BUTTON_NO) { if (buttons_count > BUTTON_NO && buttons[BUTTON_NO] == GLFW_PRESS) io.NavInputs[NAV_NO] = 1.0f; }
205  #define MAP_ANALOG(NAV_NO, AXIS_NO, V0, V1) { float v = (axes_count > AXIS_NO) ? axes[AXIS_NO] : V0; v = (v - V0) / (V1 - V0); if (v > 1.0f) v = 1.0f; if (io.NavInputs[NAV_NO] < v) io.NavInputs[NAV_NO] = v; }
206  int axes_count = 0, buttons_count = 0;
207  const float* axes = glfwGetJoystickAxes(GLFW_JOYSTICK_1, &axes_count);
208  const unsigned char* buttons = glfwGetJoystickButtons(GLFW_JOYSTICK_1, &buttons_count);
209  MAP_BUTTON(ImGuiNavInput_Activate, 0); // Cross / A
210  MAP_BUTTON(ImGuiNavInput_Cancel, 1); // Circle / B
211  MAP_BUTTON(ImGuiNavInput_Menu, 2); // Square / X
212  MAP_BUTTON(ImGuiNavInput_Input, 3); // Triangle / Y
213  MAP_BUTTON(ImGuiNavInput_DpadLeft, 13); // D-Pad Left
214  MAP_BUTTON(ImGuiNavInput_DpadRight, 11); // D-Pad Right
215  MAP_BUTTON(ImGuiNavInput_DpadUp, 10); // D-Pad Up
216  MAP_BUTTON(ImGuiNavInput_DpadDown, 12); // D-Pad Down
217  MAP_BUTTON(ImGuiNavInput_FocusPrev, 4); // L1 / LB
218  MAP_BUTTON(ImGuiNavInput_FocusNext, 5); // R1 / RB
219  MAP_BUTTON(ImGuiNavInput_TweakSlow, 4); // L1 / LB
220  MAP_BUTTON(ImGuiNavInput_TweakFast, 5); // R1 / RB
221  MAP_ANALOG(ImGuiNavInput_LStickLeft, 0, -0.3f, -0.9f);
222  MAP_ANALOG(ImGuiNavInput_LStickRight, 0, +0.3f, +0.9f);
223  MAP_ANALOG(ImGuiNavInput_LStickUp, 1, +0.3f, +0.9f);
224  MAP_ANALOG(ImGuiNavInput_LStickDown, 1, -0.3f, -0.9f);
225  #undef MAP_BUTTON
226  #undef MAP_ANALOG
227  }
228 
229  // Start the frame. This call will update the io.WantCaptureMouse, io.WantCaptureKeyboard flag that you can use to dispatch inputs (or not) to your application.
230  ImGui::NewFrame();
231  }
