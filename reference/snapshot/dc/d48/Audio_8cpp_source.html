34  int audioOutputCallback(const void* inputBuffer, void* outputBuffer,
35  size_t framesPerBuffer, const PaStreamCallbackTimeInfo* timeInfo,
36  PaStreamCallbackFlags statusFlags, void* userData)
37  {
38  // Gets the output buffer (it's of type paFloat32), and the audioStream and audioInfo
39  float* output = (float*)outputBuffer;
40  AudioOStream *audioStream = (AudioOStream*)userData;
41  AudioInfo *audioInfo = &(audioStream->audioInfo);
42 
43  // Reads the audio
44  size_t readFrames = 0, offset = 0, framesToRead = framesPerBuffer;
45  do {
46  int readFramesNow;
47  if(audioStream->_buffer != nullptr) { // Buffered
48  readFramesNow = (framesToRead + audioStream->_streamPosFrame) > audioInfo->getFrames()
49  ? audioInfo->getFrames() - audioStream->_streamPosFrame
50  : framesToRead;
51 
52  for(size_t f = 0; f < readFramesNow; f++) {
53  for(uint8_t c = 0; c < audioInfo->getChannels(); c++) {
54  output[(offset * audioInfo->getChannels()) + f * audioInfo->getChannels() + c] = audioStream->_buffer[audioStream->_streamPosFrame * audioInfo->getChannels() + (f * audioInfo->getChannels() + c)];
55  }
56  }
57  } else { // Streaming
58  readFramesNow = sf_readf_float(audioInfo->_sndFile, output + (offset * audioInfo->getChannels()), framesToRead);
59  }
60 
61  audioStream->_streamPosFrame += readFramesNow;
62  framesToRead -= readFramesNow;
63  readFrames += readFramesNow;
64 
65  if(framesToRead > 0 && audioStream->audioPlayMode == AudioPlayMode::Loop) {
66  audioStream->_streamPosFrame = 0;
67  audioStream->_loops++;
68  offset += readFramesNow;
69  if(audioStream->_buffer == nullptr) { // Streaming
70  sf_seek(audioInfo->_sndFile, 0, SF_SEEK_SET);
71  }
72  }
73  } while(framesToRead > 0 && audioStream->audioPlayMode != AudioPlayMode::Once);
74  // Adjusts the volume of each frame
75  for(size_t i = 0; i < readFrames; i++) {
76  // Applies the volume to all channels the sound might have
77  for(uint8_t channels = 0; channels < audioInfo->getChannels(); channels++) {
78  float frame = *output;
79 
80  // In case there's 3D audio, calculates 3D audio
81  if(audioStream->_audioSource != nullptr) {
82  AudioSource* source = audioStream->_audioSource;
83  const float panning = source->getPanning();
84  if(channels == 0)
85  frame *= (-0.5f * panning + 0.5f);
86  else if(channels == 1)
87  frame *= (0.5f * panning + 0.5f);
88  }
89 
90  frame *= audioStream->volume * AudioBackend::getInstance().globalVolume;
91 
92  *output++ = frame;
93  }
94  }
95 
96  // If the read frames didn't fill the buffer to read, it reached EOF
97  if(readFrames < framesPerBuffer && audioStream->audioPlayMode == AudioPlayMode::Once)
98  return paComplete;
99 
100  return paContinue;
101  }

112  int debugCallback(const void *inputBuffer, void *outputBuffer,
113  size_t framesPerBuffer, const PaStreamCallbackTimeInfo *timeInfo,
114  PaStreamCallbackFlags statusFlags, void *userData)
115  {
116 
117  uint8_t left_ear = 0;

120  uint8_t *output = (uint8_t*)outputBuffer;
121 
122  for(unsigned int i = 0; i < framesPerBuffer; i++) {

213  void AudioSource::_calculatePan()
214  {
215  Math::Vector3D listenerPos = AudioListener::getInstance().position;
216  Math::Vector3D listenerDir = AudioListener::getInstance().direction;
217  // TODO: Right now it's hardcoded because the camera, under normal circunstances,
218  // doesn't "tilt". Make less dirty later
219  Math::Vector3D listenerUp = Math::Vector3D(0, 1, 0);
220  // FIXME: The method to obtain the perpendicular to both dir and lookUp didn't use matrixes
221  // , because I am inexperienced on it. This is where I got the formula:
222  // https://math.stackexchange.com/questions/501949/determining-a-perpendicular-vector-to-two-given-vectors
223  // however, this is dirty and should be cleaned up
224  float x = listenerDir.y * listenerUp.z - listenerDir.z * listenerUp.y;
225  float y = listenerDir.z * listenerUp.x - listenerDir.x * listenerUp.z;
226  float z = listenerDir.x * listenerUp.y - listenerDir.y * listenerUp.x;
227 
228  Math::Vector3D cross = Math::Vector3D(x, y, z);
229  cross.normalize();
230 
231  // Makes a copy of position, since it needs to be manipulated to calculate the panning
232  Math::Vector3D sourcePos = _position;
233  sourcePos -= listenerPos;
234 
235  _pan = cross.dot(sourcePos.normalized());
236  }

257  : audioInfo(), _audioSource(audioSource)

262 
263  // If the audio should be buffered, do so
264  if(buffered) {
265  AuroraFW::DebugManager::Log("Buffering the audio... (Total frames: ", audioInfo.getFrames() * audioInfo.getChannels(), ")");
266  _buffer = AFW_NEW float[audioInfo.getFrames() * audioInfo.getChannels()];
267  sf_readf_float(audioInfo._sndFile, _buffer, audioInfo.getFrames());
268  AuroraFW::DebugManager::Log("Buffering complete.");
269  }

272  if(audioInfo._sndFile == nullptr)
