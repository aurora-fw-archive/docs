33  // audioCallBack
34  int audioCallback(const void *inputBuffer, void *outputBuffer,
35  unsigned long framesPerBuffer, const PaStreamCallbackTimeInfo *timeInfo,
36  PaStreamCallbackFlags statusFlags, void *userData)
37  {
38  // Gets the output buffer (it's of type paInt32)
39  int *output = (int*)outputBuffer;
40  AudioStream *audioStream = (AudioStream*)userData;
41 
42  // In case the audio stream is paused, saves the current position and stops the stream
43  if(audioStream->_audioStatus == AudioStatus::Pause) {
44  audioStream->_streamPosFrame = sf_seek(audioStream->_soundFile, 0, SF_SEEK_CUR);
45  return paComplete;
46  }
47 
48  // Reads the frames from music file (fills the buffer basically)
49  int readFrames = sf_readf_int(audioStream->_soundFile, output, framesPerBuffer);
50 
51  // Adjusts the volume of each frame
52  for(unsigned int i = 0; i < readFrames; i++) {
53  // Applies the volume to all channels the sound might have
54  for(uint8_t channels = 0; channels < audioStream->_sndInfo.channels; channels++) {
55  float frame = *output;
56 
57  // In case there's 3D audio, calculates 3D audio
58  if(audioStream->_audioSource != nullptr) {
59  AudioSource* source = audioStream->_audioSource;
60  const float panning = source->getPanning();
61  if(channels == 0)
62  frame *= (-0.5f * panning + 0.5f);
63  else if(channels == 1)
64  frame *= (0.5f * panning + 0.5f);
65  }
66 
67  frame *= audioStream->volume;
68 
69  *output++ = frame;
70  }
71  }
72 
73  // If the read frames filled the buffer to read, continues playing
74  if(readFrames == framesPerBuffer) {
75  return paContinue;
76  // Else it means it reached end of file. Stops the callback.
77  } else {
78  audioStream->_audioStatus = AudioStatus::CallbackStop;
79  return paComplete;
80  }
81  }

136  void AudioSource::calculateValues()
137  {
138  Math::Vector3D listenerPos = AudioListener::getInstance().position;
139  Math::Vector3D listenerDir = AudioListener::getInstance().direction;
140  // TODO: Right now it's hardcoded because the camera, under normal circunstances,
141  // doesn't "tilt". Make less dirty later
142  Math::Vector3D listenerUp = Math::Vector3D(0, 1, 0);
143  // FIXME: The method to obtain the perpendicular to both dir and lookUp didn't use matrixes
144  // , because I am inexperienced on it. This is where I got the formula:
145  // https://math.stackexchange.com/questions/501949/determining-a-perpendicular-vector-to-two-given-vectors
146  // however, this is dirty and should be cleaned up
147  float x = listenerDir.y * listenerUp.z - listenerDir.z * listenerUp.y;
148 
149  float y = listenerDir.z * listenerUp.x - listenerDir.x * listenerUp.z;
150 
151  float z = listenerDir.x * listenerUp.y - listenerDir.y * listenerUp.x;
152 
153  Math::Vector3D cross = Math::Vector3D(x, y, z);
154  cross.normalize();
155 
156  // Makes a copy of position, since it needs to be manipulated to calculate the panning
157  Math::Vector3D sourcePos = _position;
158  sourcePos -= listenerPos;
159 
160  _pan = cross.dot(sourcePos.normalized());
161  }

165  : _audioSource(audioSource)
166  {
167  // If path is null, run the debug callback
168  if(path == nullptr || std::string(path) == "") {
169  AuroraFW::Debug::Log("Debug mode activated for AudioStream instance");
170  getPAError(Pa_OpenDefaultStream(&_paStream, 0, 2, paUInt8,

172  } else {
173  // Gets the SNDFILE* from libsndfile

181

183  getPAError(Pa_OpenDefaultStream(&_paStream, 0, _sndInfo.channels, paInt32,
