48  size_t asize = size + sizeof(size_t);
49  byte_t* ret = (byte_t*)AFW_ALIGNED_ALLOC(asize, 2 * sizeof(size_t));

51  if(ret == AFW_NULLPTR)
52  std::cerr << "=##=\tbad alloc" << std::endl;
53  memset(ret, 0, asize);
