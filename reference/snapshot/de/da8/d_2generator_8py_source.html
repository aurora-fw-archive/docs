507  def generate_loader(self, features, extensions):
508  f = self._f_loader
509 
510  rfeatures = features
511  if self.spec.NAME in ('egl', 'wgl'):
512  features = {'egl': [], 'wgl': []}
513 
514  self.write_module(f, self.LOADER)
515  self.write_imports(f, [self.FUNCS, self.EXT, self.ENUMS, self.TYPES])
516 
517  self.loader.write(f)
518  self.loader.write_has_ext(f)
519 
520  written = set()
521  for api in self.api.items():
522  loadername = 'Load' if self.LOAD_GL_PREFIX else 'load'
523  f.write('bool {}{}{}(Loader load) {{\n'
524  .format(self.LOAD_GL_PREFIX, loadername, api.upper()))
525  self.loader.write_begin_load(f)
526  f.write('\tfind_core{}();\n'.format(api.upper()))
527  for feature in features[api]:
528  f.write('\tload_{}(load);\n'.format(feature.name))
529  f.write('\n\tfind_extensions{}();\n'.format(api.upper()))
530  for ext in extensions[api]:
531  if not list(ext.functions):
532  continue
533  f.write('\tload_{}(load);\n'.format(ext.name))
534  self.loader.write_end_load(f)
535  f.write('}\n\n')
536 
537  f.write('private {\n\n')
538 
539  f.write('void find_core{}() {{\n'.format(api.upper()))
540  self.loader.write_find_core(f)
541  if self.spec.NAME == 'gl':
542  for feature in features[api]:
543  f.write('\t{} = (major == {num[0]} && minor >= {num[1]}) ||'
544  ' major > {num[0]};\n'.format(feature.name, num=feature.number))
545  f.write('\treturn;\n')
546  f.write('}\n\n')
547 
548  f.write('void find_extensions{}() {{\n'.format(api.upper()))
549  if self.spec.NAME == 'gl':
550  for ext in extensions[api]:
551  f.write('\t{0} = has_ext("{0}");\n'.format(ext.name))
552  f.write('\treturn;\n')
553  f.write('}\n\n')
554 
555  for feature in features[api]:
556  f.write('void load_{}(Loader load) {{\n'
557  .format(feature.name))
558  if self.spec.NAME == 'gl':
559  f.write('\tif(!{}) return;\n'.format(feature.name))
560  for func in feature.functions:
561  f.write('\t{name} = cast(typeof({name}))load("{name}");\n'
562  .format(name=func.proto.name))
563  f.write('\treturn;\n}\n\n')
564 
565  for ext in extensions[api]:
566  if len(list(ext.functions)) == 0 or ext.name in written:
567  continue
568 
569  f.write('void load_{}(Loader load) {{\n'
570  .format(ext.name))
571  if self.spec.NAME == 'gl':
572  f.write('\tif(!{}) return;\n'.format(ext.name))
573  for func in ext.functions:
574  # even if they were in written we need to load it
575  f.write('\t{name} = cast(typeof({name}))load("{name}");\n'
576  .format(name=func.proto.name))
577  f.write('\treturn;\n')
578  f.write('}\n')
579 
580  written.add(ext.name)
581 
582  f.write('\n} /* private */\n\n')
