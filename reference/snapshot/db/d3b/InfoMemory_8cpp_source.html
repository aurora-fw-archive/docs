193  size_t getCurrentRSS()
194  {
195  #ifdef AFW_TARGET_PLATFORM_WINDOWS
196  PROCESS_MEMORY_COUNTERS info;
197  GetProcessMemoryInfo(GetCurrentProcess(), &info, sizeof(info));
198  return info.WorkingSetSize;
199  #elif defined(AFW_TARGET_KERNEL_LINUX)
200  FILE* fp = fopen("/proc/self/status", "r");
201  if(fp == AFW_NULL)
202  return AFW_NULLVAL;
203  size_t ret;
204  if (fscanf(fp, "%*s%lu", &ret) != 1) {
205  fclose(fp);
206  return AFW_NULLVAL;
207  }
208  fclose(fp);
209  return ret * sysconf(_SC_PAGESIZE);
210  #elif defined(AFW_TARGET_PLATFORM_APPLE_MAC)
211  struct mach_task_basic_info info;
212  mach_msg_type_number_t infoCount = MACH_TASK_BASIC_INFO_COUNT;
213  if (task_info(mach_task_self(), MACH_TASK_BASIC_INFO, (task_info_t)&info, &infoCount) != KERN_SUCCESS)
214  return AFW_NULLVAL;
215  return info.resident_size;
216  #else
217  return AFW_NULLVAL;
218  #endif
219  }
