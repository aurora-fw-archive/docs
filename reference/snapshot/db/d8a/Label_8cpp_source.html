98  void Label::setAlignment(AlignMode mode)
99  {

103  case AlignMode::TopLeft :

106  gtk_label_set_yalign((GtkLabel*)_label, 0);
107  break;
108  case AlignMode::TopCenter :
109  CLI::Log(CLI::Debug, "setting alignment mode to 'TopCenter' property from label");
110  gtk_label_set_xalign((GtkLabel*)_label, 0.5);
111  gtk_label_set_yalign((GtkLabel*)_label, 0);
112  break;
113  case AlignMode::TopRight :
114  CLI::Log(CLI::Debug, "setting alignment mode to 'TopRight' property from label");
115  gtk_label_set_xalign((GtkLabel*)_label, 1);
116  gtk_label_set_yalign((GtkLabel*)_label, 0);
117  break;
118  case AlignMode::Left :
119  CLI::Log(CLI::Debug, "setting alignment mode to 'Left' property from label");
120  gtk_label_set_xalign((GtkLabel*)_label, 0);
121  gtk_label_set_yalign((GtkLabel*)_label, 0.5);
122  break;
123  case AlignMode::Center :
124  CLI::Log(CLI::Debug, "setting alignment mode to 'Center' property from label");
125  gtk_label_set_xalign((GtkLabel*)_label, 0.5);
126  gtk_label_set_yalign((GtkLabel*)_label, 0.5);
127  break;
128  case AlignMode::Right :
129  CLI::Log(CLI::Debug, "setting alignment mode to 'Right' property from label");
130  gtk_label_set_xalign((GtkLabel*)_label, 1);
131  gtk_label_set_yalign((GtkLabel*)_label, 0.5);
132  break;
133  case AlignMode::BottomLeft :
134  CLI::Log(CLI::Debug, "setting alignment mode to 'BottomLeft' property from label");
135  gtk_label_set_xalign((GtkLabel*)_label, 0);
136  gtk_label_set_yalign((GtkLabel*)_label, 1);
137  break;
138  case AlignMode::BottomCenter :
139  CLI::Log(CLI::Debug, "setting alignment mode to 'BottomCenter' property from label");
140  gtk_label_set_xalign((GtkLabel*)_label, 0.5);
141  gtk_label_set_yalign((GtkLabel*)_label, 1);
142  break;
143  case AlignMode::BottomRight :
144  CLI::Log(CLI::Debug, "setting alignment mode to 'BottomRight' property from label");
145  gtk_label_set_xalign((GtkLabel*)_label, 1);
146  gtk_label_set_yalign((GtkLabel*)_label, 1);
147  break;
148  case AlignMode::Custom :

165  AlignMode Label::getAlignment() const
166  {

168  const float y = gtk_label_get_yalign((GtkLabel*)_label);

172  if(x == 0.0f) return AlignMode::TopLeft;

174  else if(x == 1.0f) return AlignMode::TopRight;
175  }
176  else if (y == 0.5f)

182  else if (y == 1.0f)
